public function cetak_range($mulai, $sampai)
    {
        $query =  "SELECT persediaan_progres.*,users.nama FROM " . $this->tablep . " JOIN users ON users.nip=persediaan_progres.nip where persediaan_progres.tanggal >= '$mulai' and persediaan_progres.tanggal <='$sampai' ORDER BY id DESC";
        $this->db->query($query);
        return $this->db->resultSet();
    }

controller :
	public function hapus_pengajuan_izin($id)
    {
        if ($id !== false && $id > 0) {
            $result = $this->Mpresensi->hapus_pengajuan_izin($id);
            if ($result) {
                $response = array('status' => 'success', 'message' => 'Data berhasil dihapus.');
            } else {
                $response = array('status' => 'error', 'message' => 'Gagal menghapus data.');
            }
        } else {
            $response = array('status' => 'error', 'message' => 'ID tidak valid.');
        }
        echo json_encode($response);
    }

model :
	public function hapus_pengajuan_izin($id)
    {
        $sql = "DELETE from tmp_izin where id=:id";
        $this->db->query($sql);
        $this->db->bind('id', $id);
        $this->db->execute();
        return true;
    }

model :
    function permohonan_sertifikasi($status)
    {
        $this->db->select('*');
        $this->db->from('pemohon_sertifikasi a');
        $this->db->join('perusahaan b', 'a.nama_perusahaan_sertifikasi=b.id_perusahaan', 'left');
        $this->db->join('ser_checklist c', 'a.id_pemohon_sertifikasi=c.id_pemohon_sertifikasi_checklist', 'left');
        $this->db->where('a.status_sertifikasi = "Approved"');
        if ($status == 'Belum') {
            $this->db->where('(a.tgl_validasi_cs <> "0000-00-00" or a.tgl_validasi_cs IS NOT NULL) ');
            $this->db->where('(a.tgl_validasi_tim_sertifikasi = "0000-00-00 00:00:00" or tgl_validasi_tim_sertifikasi IS NULL)');
        } else {
            $this->db->where('a.tgl_validasi_tim_sertifikasi <> "0000-00-00 00:00:00"');
        }
        $query = $this->db->get();
        if ($query->num_rows() != 0) {
            return $query->result_array();
        } else {
            return false;
        }
    }
