        allowfullscreen>
    </iframe>

<script>

// =======================
// Koordinat Sekolah
// =======================
const schoolLat = -3.434110;
const schoolLng = 114.842550;
const maxDistance = 100; // meter

let userLat = null;
let userLng = null;
let method = "";

// =======================
// Fungsi Haversine
// =======================
function getDistance(lat1, lon1, lat2, lon2) {
    const R = 6371;
    const dLat = (lat2 - lat1) * Math.PI / 180;
    const dLon = (lon2 - lon1) * Math.PI / 180;
    const a =
        Math.sin(dLat / 2) * Math.sin(dLat / 2) +
        Math.cos(lat1 * Math.PI / 180) *
        Math.cos(lat2 * Math.PI / 180) *
        Math.sin(dLon / 2) * Math.sin(dLon / 2);
    const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
    return R * c * 1000;
}

// =======================
// Set Status Banner
// =======================
function setStatus(msg, type = "info") {
    const el = document.getElementById("status");
    el.className = `alert alert-${type} mt-3`;
    el.innerHTML = msg;
}

// =======================
// Tampilkan Lokasi
// =======================
function tampilkanLokasi() {
    const distance = getDistance(userLat, userLng, schoolLat, schoolLng);

    const mapFrame = document.getElementById("mapFrame");
    mapFrame.src = `https://www.google.com/maps?q=${userLat},${userLng}&z=17&output=embed`;
    mapFrame.style.display = "block";

    setStatus(`
        📍 <strong>Metode:</strong> ${method} <br>
        🌐 <strong>Latitude:</strong> ${userLat} <br>
        🌐 <strong>Longitude:</strong> ${userLng} <br>
        📏 <strong>Jarak ke sekolah:</strong> ${distance.toFixed(2)} meter
    `, "info");
}

// =======================
// Coba satu service IP dengan timeout
// =======================
async function tryService(url, parse) {
    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 5000);
    try {
        const res = await fetch(url, { signal: controller.signal });
        clearTimeout(timeout);
        if (!res.ok) throw new Error(`HTTP ${res.status}`);
        const data = await res.json();
        return parse(data);
    } catch (e) {
        clearTimeout(timeout);
        throw e;
    }
}

// =======================
// Fallback IP Location
// =======================
async function fallbackIP() {
    setStatus("⏳ GPS tidak tersedia, mencoba lokasi via IP...", "warning");

    const services = [
        {
            // Catatan: ip-api.com hanya support HTTP (bukan HTTPS) untuk free tier
            url: "http://ip-api.com/json/?fields=status,lat,lon,city,query",
            parse: (d) => {
                if (d.status !== "success") throw new Error("Gagal");
                return { lat: d.lat, lng: d.lon, city: d.city, ip: d.query };
            }
        },
        {
            url: "https://freeipapi.com/api/json",
            parse: (d) => {
                if (!d.latitude || !d.longitude) throw new Error("Gagal");
                return { lat: d.latitude, lng: d.longitude, city: d.cityName, ip: d.ipAddress };
            }
        },
        {
            url: "https://ipwho.is/",
            parse: (d) => {
                if (!d.success) throw new Error("Gagal");
                return { lat: d.latitude, lng: d.longitude, city: d.city, ip: d.ip };
            }
        },
        {
            url: "https://ipinfo.io/json",
            parse: (d) => {
                if (!d.loc) throw new Error("Gagal");
                const [lat, lng] = d.loc.split(",").map(Number);
                return { lat, lng, city: d.city, ip: d.ip };
            }
        }
    ];

    for (let i = 0; i < services.length; i++) {
        const s = services[i];
        setStatus(`⏳ Mencoba service IP (${i + 1}/${services.length})...`, "warning");
        try {
            const loc = await tryService(s.url, s.parse);

            userLat = parseFloat(loc.lat);
            userLng = parseFloat(loc.lng);
            method = `IP Location — ${loc.ip || ''} (${loc.city || ''})`;

            tampilkanLokasi();
            return;

        } catch (err) {
            console.warn(`Service [${s.url}] gagal:`, err.message);
        }
    }

    // Semua gagal — tawarkan input manual
    setStatus(`
        ❌ <strong>Semua metode lokasi gagal.</strong><br>
        <small>Kemungkinan penyebab: koneksi bermasalah, browser memblokir HTTP request, atau semua service IP tidak tersedia.</small><br><br>
        <strong>Masukkan koordinat secara manual:</strong><br>
        <div class="d-flex gap-2 mt-2 flex-wrap">
            <input type="number" id="manualLat" class="form-control w-auto" placeholder="Latitude" step="any">
            <input type="number" id="manualLng" class="form-control w-auto" placeholder="Longitude" step="any">
            <button class="btn btn-sm btn-success" onclick="inputManual()">Gunakan</button>
            <button class="btn btn-sm btn-outline-secondary" onclick="fallbackIP()">🔄 Coba Lagi</button>
        </div>
    `, "danger");
}

// =======================
// Input Lokasi Manual
// =======================
function inputManual() {
    const lat = parseFloat(document.getElementById("manualLat").value);
    const lng = parseFloat(document.getElementById("manualLng").value);

    if (isNaN(lat) || isNaN(lng)) {
        alert("⚠️ Masukkan koordinat yang valid.");
        return;
    }

    userLat = lat;
    userLng = lng;
    method = "Input Manual";
    tampilkanLokasi();
}

// =======================
// Ambil Lokasi Saat Load
// =======================
window.onload = function () {

    if (!navigator.geolocation) {
        fallbackIP();
        return;
    }

    navigator.geolocation.getCurrentPosition(
        function (position) {
            userLat = position.coords.latitude;
            userLng = position.coords.longitude;
            method = "GPS / WiFi Positioning";
            tampilkanLokasi();
        },
        function (error) {
            console.warn("Geolocation error:", error.code, error.message);
            fallbackIP();
        },
        {
            timeout: 8000,
            maximumAge: 60000,
            enableHighAccuracy: false
        }
    );
};

// =======================
// Tombol Absensi
// =======================
function absen() {

    if (userLat === null) {
        alert("⚠️ Lokasi belum tersedia. Harap tunggu atau masukkan koordinat manual.");
        return;
    }

    const distance = getDistance(userLat, userLng, schoolLat, schoolLng);
    const resultEl = document.getElementById("result");

    if (distance <= maxDistance) {
        resultEl.innerHTML = `
            <div class="alert alert-success mt-2">
                ✅ <strong>Absensi Berhasil!</strong><br>
                Jarak Anda dari sekolah: <strong>${distance.toFixed(2)} meter</strong>
            </div>
        `;
    } else {
        resultEl.innerHTML = `
            <div class="alert alert-danger mt-2">
                ❌ <strong>Absensi Gagal</strong> — Anda di luar area sekolah.<br>
                Jarak Anda: <strong>${distance.toFixed(2)} meter</strong> (maks. ${maxDistance} meter)
            </div>
        `;
    }
}
